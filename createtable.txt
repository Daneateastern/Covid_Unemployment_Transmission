	FIPS VARCHAR (50),

	year VARCHAR (50),
	month VARCHAR ( 50 ),
	unemployment_Total FLOAT (50),
      unemployment_Rate FLOAT (10),
	date_key VARCHAR PRIMARY KEY
);

CREATE TABLE Unemployment_Rate_By_City (
	FIPS VARCHAR (50),
	city VARCHAR ( 50 ),
	year VARCHAR (50),

	unemployment_Total FLOAT (50),
      unemployment_Rate FLOAT (10),
	date_key VARCHAR PRIMARY KEY
);
